import random
import string

def distort_token(token):
    word = token.text
    idx = random.randint(0, len(word) - 1)
    
    distortion_type = random.randint(0, 2)
    new_word = word
    if distortion_type == 0 and len(word) > 1:
        new_word =  word[:idx] + word[idx + 1:]
    elif distortion_type == 1:
        random_letter = random.choice(string.ascii_lowercase)
        new_word = word[:idx] + random_letter + word[idx:]
    elif distortion_type == 2:
        random_letter = random.choice(string.ascii_lowercase)
        new_word =  word[:idx] + random_letter + word[idx + 1:]
    return new_word

def distort_query_tokens(query):
    doc = tokenizer(query)
    tokens = list(doc)
    
    token_idx = random.randint(0, len(tokens) - 1)
    distorted_token = distort_token(tokens[token_idx])
    
    tokens[token_idx] = distorted_token
    return " ".join(str(token) for token in tokens)

def bulk_load(dataset):
    bulk_data = ""
    for entry in dataset["corpus"]:
        doc_id = entry["_id"]
        bulk_data += json.dumps({"index": {"_index": index_name, "_id": doc_id}}) + "\n"
        bulk_data += json.dumps({"text": entry['text']}) + "\n"



    bulk_response = requests.post(f"{es_url}/_bulk", headers={"Content-Type": "application/x-ndjson"}, data=bulk_data)

    if bulk_response.status_code == 200:
        response_data = bulk_response.json()
        if any(item.get("index", {}).get("error") for item in response_data["items"]):
            print("Some documents failed to index:")
            for item in response_data["items"]:
                if "error" in item["index"]:
                    print(item["index"]["error"])
        else:
            print("All documents indexed successfully.")
    else:
        print(f"Failed to index data: {bulk_response.text}")

def prepare_fiqa_qrels():
    query_to_corpus_dict = {}

    subsets = ['test', 'validation', 'train']

    for subset in subsets:
        for item in fiqa_qrels[subset]:
            if item['query-id'] not in query_to_corpus_dict:
                query_to_corpus_dict[item['query-id']] = {}

            query_to_corpus_dict[item['query-id']][item['corpus-id']] = item['score']

    for query_id in query_to_corpus_dict:
        sorted_corpuses_by_score = dict(sorted(query_to_corpus_dict[query_id].items(), key=lambda item: item[1]))
        query_to_corpus_dict[query_id] = sorted_corpuses_by_score

    return query_to_corpus_dict

def prepare_existing_fiqa_queries(query_to_corpus_dict):
    queries_dict = {}

    queries_dataset = fiqa_queries['queries']
    for entry in queries_dataset:
        if int(entry['_id']) in query_to_corpus_dict.keys():
            queries_dict[int(entry['_id'])] = entry['text']

    return queries_dict

dcg_limit = 10

def find_for_phrase(search_phrase, search_field, custom_analyzer, fuzzy):
    search_query = {
        "size": dcg_limit,
        "query": {
            "match": {
                search_field: {
                    "query": search_phrase,
                    "fuzziness": 'AUTO' if fuzzy else 0
                }
            }
        }
    }

    if custom_analyzer:
        search_query['query']['match'][search_field]['analyzer'] = custom_analyzer

    response = requests.get(f"{index_url}/_search", headers={"Content-Type": "application/json"}, data=json.dumps(search_query))

    if response.status_code == 200:
        search_results = response.json()
        return dict(list(map(lambda hit: (int(hit['_id']), float(hit['_score'])), search_results["hits"]["hits"])))
    else:
        print(f"Search failed: {response.text}")

import math

relevant_doc_number = 0
relevant_docs = []

def calculate_dcg(docs, docs_scoring):
    sum = 0
    relevant_doc_id = None
    for i, doc_id in enumerate(docs):
        if doc_id in docs_scoring.keys():
            sum += (2**docs_scoring[doc_id]-1)/(math.log2(i+1+1))
            if i == relevant_doc_number:
                relevant_doc_id = doc_id
    return sum, relevant_doc_id

def calculate_ndcgs(queries, search_field, custom_analyzer, fuzzy):
    
    ndcgs = []

    for query in queries:
        q_id = int(query['_id'])
        q_text = query['text']
        if(q_id not in queries_dict.keys()):
            continue
        ideal_search = list(query_to_corpus_dict[q_id].keys())[:dcg_limit]
        idcg, _ = calculate_dcg(ideal_search, query_to_corpus_dict[q_id])

        real_search_with_scores = find_for_phrase(q_text, search_field, custom_analyzer, fuzzy)
        dcg, relevant_doc_id = calculate_dcg(real_search_with_scores.keys(), query_to_corpus_dict[q_id])

        ndcgs.append(dcg/idcg)

        if relevant_doc_id:
            relevant_docs.append((relevant_doc_id, real_search_with_scores[relevant_doc_id], real_search_with_scores.values(), q_text))

    return ndcgs

def read_morfeusz_into_set(file_path):
    correct_words = set()

    with open(file_path, "r", encoding="utf-8") as file:
        for line_number, line in enumerate(file):
            if line_number < 31:
                continue
            word = line.split("\t")[0]
            correct_words.add(word.lower())
    return correct_words

import re

def split_to_words(text): return re.findall(r'\w+', text)

def P(word, N=sum(token_counter.values())): 
    "Probability of `word`."
    return token_counter[word] / N

def correct(word): 
    return max(candidates(word), key=P)

def candidates(word): 
    "Generate possible spelling corrections for word."
    return (known([word]) or known(edits1(word))  or [word])

def known(words): 
    "The subset of `words` that appear in the dictionary of WORDS."
    return set(w for w in words if w in correct_words)

def edits1(word):
    "All edits that are one edit away from `word`."
    letters    = 'abcdefghijklmnopqrstuvwxyz'
    splits     = [(word[:i], word[i:])    for i in range(len(word) + 1)]
    deletes    = [L + R[1:]               for L, R in splits if R]
    transposes = [L + R[1] + R[0] + R[2:] for L, R in splits if len(R)>1]
    replaces   = [L + c + R[1:]           for L, R in splits if R for c in letters]
    inserts    = [L + c + R               for L, R in splits for c in letters]
    return set(deletes + transposes + replaces + inserts)

skip_word_pattern = re.compile(r"(\d|_)+")

def correct_queries_text(query):
    words = split_to_words(query)
    corrected_query = query
    for word in words:
        low_word = word.lower()
        if low_word not in correct_words:
            if skip_word_pattern.match(low_word):
                continue
            corrected_word = correct(low_word)
            corrected_query = corrected_query.replace(word, corrected_word)
    return corrected_query

Wnioski po wykonaniu wiczenia na temat:

1. Rozkadu sw w korpusie:

Przegldajc listy frekwencji mona zauway dominujc liczb nastpujcych czci mowy:

  • przyimki: w, na, z ...
  • spjniki: i, e, czy, lub, aby ...
  • partykuy: nie, czy, tylko ...
  • zaimki: si, co, ktre ...

Wystpuj rwnie czasowniki, rzeczowniki, przymiotniki itp. lecz s one znacznie rzadsze od wspomnianych powyej czci mowy, wic plasuj si odpowiednio niej. Nielicznymi wyjtkami wrd czasownikw s: s, moe, by.

Obserwacja wystpujcych form mowy jest bardzo wana dla jakoci wynikw w metryce NDCG@10. Widzc tak odlege miejsca czci mowy jak rzeczowniki, czasowniki, czy te przymiotniki, ktore to nios wikoszo informacji na temat naszego zapytania, nie dziwi fakt, e skuteczno dopasowanych wynikw jest do niska, gdy odnalezione zostanie mnstwo dokumentw pasujcych poprzez formach mowy typu przyimek, spjnik, partykua. S one do czste, lecz nie nios ze sob duo istotnej informacji.

Podobnie jak w poprzednim wiczeniu, rwnie i tym razem wskazuj, e wprowadzenie filtra typu stopwords mogoby w znacznym stopniu poprawi jako dopasowa zwracanych przez indeks.

2. Szybko wykonania zapyta:

Porwnanie wydajnoci wykonam dla dwch przypadkw, w kaym z nich zmierz czas sumaryczny dla wszystkich wywoa Elasticsearch, ktrych jest tyle, ile krotek w zbiorze queries FIQA-PL.

  • lolakla poprawa zapyta z uyciem funkcji opartej o odlego Levenstheina + zapytanie o bezporednie dopasowanie do Elasticsearch z filtrem morfologik: 0:01 + 1:05 = 1:06
  • zapytanie z zastosowaniem fuzzy-search do Elasticsearch z filtrem morfologik bez uprzedniej poprawy zapyta zawierajcych bdy: 3:54

Jak wida zapytanie z uyciem fuzzy-search potrzebuje niemale cztery razy wicej czasu od lokalnej korekcji wyrae i zapytania z dopasowaniem bezporednim.

Dodam jeszcze tylko, e w sumie wszystkich zapyta do Elasticsearch w kadym z podpunktw jest 6648, a wic na pojedycze zapytanie potrzeba odpowiednio <0.01s i 0.035, a wic w obydwu przypadkach odpowied na pojedycze zapytanie do serwera jest zwracana w zaniedbywalnym dla ludzkiej percepcji.

3. Jako uzyskanych wynikw:

Jako wynikw pokazuje bardzo ciekaw zaleno, a mianowicie e poprawa literwek w zapytaniu doprowadzia do obnienia wartoci metryki NDCG@10, i to dla obydwu podej. Widz dla tego stanu rzeczy nastpujce wyjanienie:

  • przy korekcji funkcj lokaln i zapytaniu z dopasowaniem bezporednim:
    opierajc si na rozkadzie form mowy w zbiorze, ktry to pokazuje na znaczn przewag wyrazw nie wnoszcych istotnych informacji, wprowadzenie literw w duej mierze musiao rwnie dotyczy takich wanie sabych jakociowo wyrazw, przez co bezporednie dopasowanie nie uwzgldniao ich i niejako automatycznie pozostae wyrazy, w tym te o duej istotnoci dla wyszukiwanej frazy, nabieray wikszego znaczenia.
    Poprawa literw przyniosaby zapewne znacz popraw przy bezporednim dopasowaniu, gdyby wczeniej zestaw danych zosta lepiej przygotowany i nie zawiera mao wnoszcych wyrazw.

  • przy zastosowaniu zapytania z fuzzy-search: w tym przypadku korekrcja nie obejmuje tylko jednego wyrazu, a wszystkich wyrazw z zapytania, a co wicej przy duszych wyrazach moe ona dopuszcza odlego Levenstheina > 1. Przy takim podejciu mnstwo swek podobnych do tych wyszukiwanych, rwnie zostaje zaakceptowanych, przez co dokadaj si do zaburzenia zwracanych wynikw.
    Generalnie dobr praktyk jest wprowadzenie boostingu, czyli swojego rodzaju wag, ktre to wyej oceniaj bezporednia dopasowania, niej natomiast dopasowania typu fuzzy. W poleceniu bya proba o zastosowanie zapyta fuzzy, std nie zaaplikowaem tej zasady.
    Poza tym, tak jak i w poprzednim punkcie, lepsze odfiltrowanie mao znaczcych sw znacznie poprawioby nasze wyniki.

Generalnie naley jeszcze zauway, e te rnice w wartoci NDCG@10 midzy zapytaniami z wyraeniami zawierajcymi literwki, a dwoma metodami radzenia sobie z nimi s do mae. Poza tym jedynie 36-37% wszystkich zapyta nie otrzymao wartoci 0 w NDCG@10, a rednia wrd tych wynikw to 0.48 +/-0.01, czyli do kiepsko.
Wszystko to wskazuje na zdecydowan potrzeb lepszego przygotowania danych do wyszukiwania.
